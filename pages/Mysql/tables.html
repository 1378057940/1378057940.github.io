emp员工表

CREATE TABLE `emp`
(
    `id`           int(10) unsigned NOT NULL AUTO_INCREMENT,
    `work_no`      varchar(10) CHARACTER SET latin1  COMMENT '工号',
    `name`         varchar(10)                       COMMENT '姓名',
    `gender`       char(1)                           COMMENT '性别',
    `age`          tinyint(3) unsigned               COMMENT '年龄',
    `job`          varchar(20)                       COMMENT '职位',
    `salary`       int(11)                           COMMENT '薪资',
    `id_card`      char(18)                          COMMENT '身份证号',
    `work_address` varchar(255)                      COMMENT '工作地址',
    `entry_date`   date                              COMMENT '入职时间',
    `managerid`    int(11)                           COMMENT '直属领导id',
    `dept_id`      int(11)                           COMMENT '部门id',
    PRIMARY KEY (`id`),
    KEY `idx_emp_work_age_name` (`work_no`, `age`, `name`) USING BTREE,
    KEY `idx_emp_work` (`work_no`) USING BTREE
) ENGINE = InnoDB AUTO_INCREMENT = 17 DEFAULT CHARSET = utf8mb4 COMMENT ='员工表'

    `dept_name` varchar(50) NOT NULL COMMENT '部门名称',
